# Calculate, and put the sum of the arithmetic series 100 + 95 + 90 + 85 + â€¦ + 5 in AX


.MODEL SMALL
.STACK 100H
.DATA
   


.CODE
MAIN PROC 
          MOV ax,@DATA 
          MOV ds,ax 
          
          mov ax,0d
          mov dx,100d
          
          start:
          cmp dx,4d
          jg gg
          jmp end
          
          gg: 
          
          add ax,dx
          
          sub dx,5d
            
          
          jmp start    
          
          
          end:

          
          
         
          
          MAIN ENDP
          END MAIN
